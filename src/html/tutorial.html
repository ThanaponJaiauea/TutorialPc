resObj() (ฟังก์ชันแปลงโค๊ดทั้งหมดเข้า base 64 สำหรับนำไปสร้างไฟล์ build app)

Example

// แปลงไฟล์ html to json function resObj(mResUrl) { var resUrl = "res.json"; var resData = null; var base64 = new Base64(); var successFun = null; var errorFun = null; this.init = function () { if (mResUrl) { resUrl = mResUrl; } } this.setSuccessFun = function (mSuccessFun) { if (mSuccessFun) { successFun = mSuccessFun; } } this.setErrorFun = function (mErrorFun) { if (mErrorFun) { errorFun = mErrorFun; } } this.load = function () { $.ajax({ type: "get", url: resUrl, contentType: "charset=utf-8", data: "", async: true, timeout: 60000, beforeSend: function (xmlhttprequest) { }, success: function (data) { //console.log(data); resData = data; if (successFun) { successFun(); } }, error: function (xmlhttprequest, error) { console.log('error===>' + resUrl + "error===>" + error + ", xmlhttprequest===>" + xmlhttprequest); if (errorFun) { errorFun(); } }, complete: function () { } }); } this.getResData = function () { return resData; } this.showHtml = function (path, divId) { if (resData != null) { var resStr = String(resData[path]); $("#" + divId).append(base64.decode(resStr)); } } }
